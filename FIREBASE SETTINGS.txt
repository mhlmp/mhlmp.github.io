rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 1. חסימה מוחלטת כברירת מחדל לכל מסד הנתונים
    match /{document=**} {
      allow read, write: if false;
    }

    // 2. הגישה למסמכי המשתמש (שם יושבים מפתחות ה-API היקרים)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

      // 3. הגנה על תת-אוסף הקישורים
      match /items/{itemId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }

      // 4. הגנה על תת-אוסף סל המחזור
      match /trash/{trashId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
  }
}
